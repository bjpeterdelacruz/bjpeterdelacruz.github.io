[docs]def sol(num):
    """
    Kata author's solution

    :param num: Number to test whether it is prime
    :return: True if the number is prime, False otherwise
    """
    return num == 2 or num % 2 and num > 2 and all(
        num % curr for curr in range(3, int(num ** .5) + 1, 2))
