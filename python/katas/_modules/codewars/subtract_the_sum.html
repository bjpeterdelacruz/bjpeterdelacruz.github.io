[docs]def subtract_sum(number: int) -> str:
    """
    This function sums all the digits in :attr:`number`. If the sum is found in the map, the name of
    the fruit that is mapped to the sum is returned. Otherwise, all the digits of the sum are summed
    together. This process is repeated until the sum is found in the map. If :attr:`number` is less
    than one, a ValueError is raised.

    :param number: The input integer
    :return: The name of a piece of fruit that is mapped to the sum of the digits of an integer
    """
    fruits_map = {int(fruit.split("-")[0]): fruit.split("-")[1] for fruit in FRUITS.split()}
    while True:
        number -= sum(int(digit) for digit in str(number))
        if number in fruits_map:
            return fruits_map[number]
